  mutate(

    wt = st_inverse_distance(nb, geometry,
                             scale = 1,
                             alpha = 1)
  )

Calculating Local Moran’s I

lmi_biz <- biz_knn6 %>% 
  mutate(local_moran = local_moran(
    BIZ_COUNT, nb, wt, nsim = 99),
         .before = 1) %>%
  unnest(local_moran)
